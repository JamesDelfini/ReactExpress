# Initialize the project to create package.json file
    Running the command below will ask the details for the your project
    > npm init
    
    # Installation
        # Express
            Node.js web application framework
            > npm i express concurrently
        
        # Nodemon
            Dev dependencies, constantly watch project changes
            > npm i nodemon --save-Dev
        
        # React
            Global Installation of React
            > npm i -g create-react-app
        
    # Scripts
        # Run the express server only
            > npm run-script server

        # Run react application only
            > cd client/
            > npm run start
